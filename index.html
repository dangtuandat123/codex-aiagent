    # Tính tổng dãy số và in kết quả
def sum_numbers(nums: list[int]) -> int:
    return sum(nums)

def fizzbuzz(n: int) -> list[str]:
    out: list[str] = []
    for i in range(1, n + 1):
        if i % 15 == 0:
            out.append("FizzBuzz")
        elif i % 3 == 0:
            out.append("Fizz")
        elif i % 5 == 0:
            out.append("Buzz")
        else:
            out.append(str(i))
    return out

if __name__ == "__main__":
    nums = [1, 2, 3, 4, 5]
    print("Sum =", sum_numbers(nums))
    print("FizzBuzz:", ", ".join(fizzbuzz(16)))
